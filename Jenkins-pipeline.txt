pipeline {
    agent any

    tools{
        allure 'allure'
    }

    stages {
        stage('Clone Repository') {
            steps {
                // Checkout the desired Git branch
                script {
                    git branch: 'jenkins-config', url: 'https://github.com/Vivekkr0311/SeleniumAutomationJava.git'
                    echo 'Checked out "jenkins-config" branch successfully'
                }
            }
        }
    }

        stage('Execute Regression Tests') {
            steps {
                // Change into the cloned repository directory
                sh 'mvn clean install'
                sh 'mvn test -DsuiteXmlFile=regression.xml'
                echo 'Regression tests executed successfully'
            }
        }
    }

    post {
        always {
            // Publish Allure reports
            allure([
                includeProperties: true,
                jdk: '',
                properties: [],
                reportBuildPolicy: 'ALWAYS',
                results: [[path: '/allure-results']]
            ])

            sh 'rm -rf allure-results'
            // Clean the workspace after the pipeline
            deleteDir()
        }
    }
}
